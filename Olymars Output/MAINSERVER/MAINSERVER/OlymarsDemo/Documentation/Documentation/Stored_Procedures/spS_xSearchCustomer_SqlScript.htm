CREATE Procedure [spS_xSearchCustomer]
(
@LastName varchar(50) = Null
,@FirstName varchar(50) = Null
)

As

Select
[CustomerDisplay].[ID1]
, [CustomerDisplay].[Display]

From tblCustomer
Inner Join [fntblCustomer_Display](Null) [CustomerDisplay] On [Cus_LngID] = [CustomerDisplay].[ID1]

Where
(@LastName Is Null Or Cus_StrLastName Like @LastName)
And (@FirstName Is Null Or Cus_StrFirstName Like @FirstName)

Order By Cus_StrLastName Asc, Cus_StrFirstName Asc

Return(@@RowCount)
