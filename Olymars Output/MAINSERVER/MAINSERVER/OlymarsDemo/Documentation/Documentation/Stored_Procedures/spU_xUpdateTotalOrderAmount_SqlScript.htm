CREATE Procedure [spU_xUpdateTotalOrderAmount]
(
@Ord_GuidID uniqueidentifier
)

As

Declare @Total Money

Select @Total = Sum(tblProduct.Pro_CurPrice * tblOrderItem.Oit_LngAmount)
From tblOrderItem Inner Join tblProduct On tblOrderItem.Oit_GuidProductID = tblProduct.Pro_GuidID
Where tblOrderItem.Oit_GuidOrderID = @Ord_GuidID
Group By tblOrderItem.Oit_GuidOrderID

Update tblOrder Set Ord_CurTotal = @Total Where Ord_GuidID = @Ord_GuidID

Return(0)
